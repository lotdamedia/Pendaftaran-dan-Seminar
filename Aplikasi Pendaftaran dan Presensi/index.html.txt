        function setupNavigation() {
            const addNav = (id, viewId) => {

        function showView(viewId) {
            document.querySelectorAll('.view-section').forEach(v => v.style.display = 'none');
            const v = document.getElementById(viewId);
            if(v) {
                v.style.display = 'block';
                v.classList.remove('animate-fade-in', 'animate-slide-up');
                void v.offsetWidth; v.classList.add('animate-fade-in');
            }
            
            // Hentikan scanner dengan aman jika admin pindah ke menu lain
            if (html5QrCode && html5QrCode.isScanning && viewId !== 'scanner-view') {
                html5QrCode.stop().then(() => { html5QrCode.clear(); }).catch(e => {}); 
                document.getElementById('scan-laser').classList.add('hidden');
            }
            
            if (viewId === 'scanner-view') {
                // Berikan jeda waktu 300ms agar DOM & Animasi ter-render dulu sebelum kamera nyala
                setTimeout(() => { startScanner(); }, 300);
            }
            if (viewId === 'participants-view') loadParticipants();
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function openModal(id) { document.getElementById(id).classList.remove('hidden'); }
        function closeModal(id) { document.getElementById(id).classList.add('hidden'); }
        function setAdminView(isAdmin) {
            document.querySelectorAll('#admin-login-btn, #admin-login-btn-mobile').forEach(e => e.classList.toggle('hidden', isAdmin));
            document.querySelectorAll('#admin-menu-btn, #admin-menu-btn-mobile').forEach(e => e.classList.toggle('hidden', !isAdmin));
        }

        function handleLogin() {
            if (document.getElementById('passkey').value === adminPasskey) {
                sessionStorage.setItem('isAdminLoggedIn', 'true'); setAdminView(true); closeModal('loginModal');
                document.getElementById('passkey').value = ''; showView('admin-view');
            } else {
                Swal.fire({icon: 'error', title: 'Akses Ditolak', text: 'Passkey salah!', background: '#1e293b', color: '#fff'});
            }
        }

            function startScanner() {
            document.getElementById('scan-laser').classList.remove('hidden');
            
            // Inisialisasi HTML5QRcode jika belum ada
            if (!html5QrCode) {
                html5QrCode = new Html5Qrcode("qr-reader");
            }
            
            // Attempt 1: Coba jalankan dengan mode kamera saat ini (Default: Kamera Belakang)
            html5QrCode.start(
                { facingMode: currentFacingMode }, 
                { fps: 10, qrbox: { width: 250, height: 250 } }, 
                onScanSuccess
            ).catch(err => { 
                console.warn("Attempt 1 Failed (" + currentFacingMode + "):", err);
                
                // Fallback: Jika pengguna memakai Laptop/PC yang tidak punya kamera belakang
                if (currentFacingMode === "environment") {
                    console.log("Mencoba fallback ke kamera depan (user)...");
                    currentFacingMode = "user"; // Ubah prioritas ke kamera depan
                    
                    html5QrCode.start(
                        { facingMode: "user" }, 
                        { fps: 10, qrbox: { width: 250, height: 250 } }, 
                        onScanSuccess
                    ).catch(err2 => showCameraError(err2));
                } else {
                    showCameraError(err);
                }
            });
        }

        function showCameraError(errorObj) {
            document.getElementById('scan-laser').classList.add('hidden'); 
            
            // Ekstrak nama error sistem
            let errorDetails = errorObj.name || errorObj.message || errorObj;
            let saran = "Pastikan izin kamera diberikan di pengaturan browser.";
            
            // Analisa penyebab error
            if (String(errorDetails).includes("NotAllowedError")) saran = "Izin kamera diblokir. Klik ikon gembok di sebelah URL dan ubah izin kamera menjadi 'Allow/Izinkan'.";
            if (String(errorDetails).includes("NotReadableError")) saran = "Kamera tidak bisa dibaca. Pastikan kamera tidak sedang dipakai oleh aplikasi lain (seperti Zoom atau Google Meet).";
            if (String(errorDetails).includes("NotFoundError")) saran = "Tidak ada satupun kamera hardware yang terdeteksi di perangkat Anda.";

            Swal.fire({
                icon: 'error', 
                title: 'Gagal Membuka Kamera', 
                html: `<p class="text-sm text-gray-300 mb-3">${saran}</p>
                       <p class="text-xs text-rose-400 bg-rose-500/10 p-2 rounded border border-rose-500/20 text-left font-mono">Log sistem: ${errorDetails}</p>`, 
                background: '#1e293b', 
                color: '#fff'
            });
        }
        function switchCamera() {
            if (html5QrCode && html5QrCode.isScanning) {
                html5QrCode.stop().then(() => { 
                    currentFacingMode = currentFacingMode === "environment" ? "user" : "environment"; 
                    startScanner(); 
                }).catch(err => console.warn(err));
            }
        }
        function onScanSuccess(decodedText) {
            if (html5QrCode && html5QrCode.isScanning) { 
                html5QrCode.stop().then(() => html5QrCode.clear()).catch(e=>{}); 
                document.getElementById('scan-laser').classList.add('hidden'); 
            }
            const type = document.querySelector('input[name="scanType"]:checked').value;

        }
        function handleManualCheckIn() {

        function onCheckInSuccess(res) {
            if (res.status === "SUCCESS") Swal.fire({icon: 'success', title: `Berhasil Absen ${res.type}`, text: res.name, background: '#1e293b', color: '#fff'});
            else if (res.status === "INFO") Swal.fire({icon: 'info', title: `Sudah Absen`, text: `${res.name} sudah absen ${res.type}`, background: '#1e293b', color: '#fff'});
            else Swal.fire({icon: 'error', title: 'Gagal', text: res.message, background: '#1e293b', color: '#fff'});
            
            setTimeout(() => { if (document.getElementById('scanner-view').style.display === 'block') startScanner(); }, 2000);
        }

        async function cetakPDF() {
            const { jsPDF } = window.jspdf;
            const content = document.getElementById('bukti-pendaftaran');
            content.style.background = 'white'; content.style.padding = '20px';
            const canvas = await html2canvas(content, { scale: 2 });
            content.style.background = ''; content.style.padding = '';
            
            const pdf = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a5' });
            pdf.addImage(canvas.toDataURL('image/png'), 'PNG', 0, 0, pdf.internal.pageSize.getWidth(), (canvas.height * pdf.internal.pageSize.getWidth()) / canvas.width);
            pdf.save(`Tiket-${document.getElementById('popup-nama').textContent}.pdf`);
        }

        async function generateSertifikatPDF(nama) {
            Swal.fire({title: 'Membuat Sertifikat...', text: 'Mohon tunggu', background: '#1e293b', color: '#fff', didOpen: () => Swal.showLoading()});
            
            document.getElementById('cert-name').textContent = nama;
            const certNode = document.getElementById('cert-canvas');
            
            // Show temporarily for html2canvas to read
            const container = document.getElementById('certificate-template');
            container.style.opacity = '1'; container.style.zIndex = '9999'; container.style.left = '0';
            
            try {
                const canvas = await html2canvas(certNode, { scale: 2, useCORS: true });
                const { jsPDF } = window.jspdf;
                const pdf = new jsPDF({ orientation: 'landscape', unit: 'mm', format: 'a4' });
                
                pdf.addImage(canvas.toDataURL('image/png'), 'PNG', 0, 0, 297, 210);
                pdf.save(`Sertifikat_${nama.replace(/\s+/g, '_')}.pdf`);
                Swal.fire({icon: 'success', title: 'Berhasil', text: 'Sertifikat diunduh!', background: '#1e293b', color: '#fff'});
            } catch (err) {
                Swal.fire({icon: 'error', title: 'Gagal', text: 'Gagal merender sertifikat.', background: '#1e293b', color: '#fff'});
            } finally {
                container.style.opacity = '0'; container.style.zIndex = '-1'; container.style.left = '-9999px';
            }
        }

        function loadParticipants() {
            document.getElementById('participants-list').innerHTML = '<div class="p-8 text-center text-gray-400"><span class="material-icons animate-spin">sync</span></div>';

        function togglePayment(phone) {
            Swal.fire({title: 'Update Pembayaran...', background: '#1e293b', color: '#fff', didOpen: () => Swal.showLoading()});
            google.script.run.withSuccessHandler(res => {
                if(res.status === "SUCCESS") {
                    const Toast = Swal.mixin({toast: true, position: 'top-end', showConfirmButton: false, timer: 3000, background: '#1e293b', color: '#fff'});
                    Toast.fire({icon: 'success', title: `${res.name} kini ${res.newStatus}`});
                    loadParticipants(); // refresh
                }
            }).updatePaymentStatus(phone);
        }

        function renderTable(data) {
            const list = document.getElementById('participants-list');
            const search = document.getElementById('searchInput').value.toLowerCase();
            
            let filtered = data;
            if (search) filtered = data.filter(r => r[1].toLowerCase().includes(search) || r[3].toLowerCase().includes(search));
            
            if (filtered.length === 0) return list.innerHTML = '<div class="p-8 text-center text-gray-400">Tidak ada data.</div>';
            
            let html = `<table class="w-full text-sm text-left text-gray-300">
                <thead class="text-xs text-gray-400 uppercase bg-gray-800/50">
                    <tr><th class="px-4 py-3">Nama & Kontak</th><th class="px-4 py-3">Instansi</th><th class="px-4 py-3 text-center">Masuk</th><th class="px-4 py-3 text-center">Pulang</th><th class="px-4 py-3 text-center">Pembayaran</th></tr>
                </thead><tbody class="divide-y divide-white/5">`;
                
            filtered.forEach(row => {
                const [waktu, nama, alamat, hp, instansi, masuk, pulang, bayar] = row;
                
                const badgeMasuk = masuk ? `<span class="bg-emerald-500/20 text-emerald-400 px-2 py-1 rounded text-xs">${masuk.split(' ')[1] || 'Hadir'}</span>` : '-';
                const badgePulang = pulang ? `<span class="bg-blue-500/20 text-blue-400 px-2 py-1 rounded text-xs">${pulang.split(' ')[1] || 'Pulang'}</span>` : '-';
                
                const isLunas = bayar === 'Lunas';
                const btnBayar = `<button onclick="togglePayment('${hp}')" class="px-3 py-1 rounded-full text-xs font-bold transition-colors ${isLunas ? 'bg-emerald-600 hover:bg-emerald-500 text-white' : 'bg-rose-600 hover:bg-rose-500 text-white'}">${isLunas ? 'LUNAS' : 'BELUM'}</button>`;
                
                html += `<tr class="hover:bg-white/5 transition-colors">
                    <td class="px-4 py-3"><div class="font-medium text-white">${nama}</div><div class="text-xs text-gray-500">${hp}</div></td>
                    <td class="px-4 py-3">${instansi}</td>
                    <td class="px-4 py-3 text-center">${badgeMasuk}</td>
                    <td class="px-4 py-3 text-center">${badgePulang}</td>
                    <td class="px-4 py-3 text-center">${btnBayar}</td>
                </tr>`;
            });
            list.innerHTML = html + `</tbody></table>`;
        }
